DirectXMath (CPU SIMD) vs HLSL (GPU SIMD)

C++ / DirectXMath

#include <DirectXMath.h>
using namespace DirectX;

// Compute 4 Fibonacci numbers in parallel
XMVECTOR FibonacciSIMD(XMVECTOR n) {
    const XMVECTOR phi =
        XMVectorReplicate(1.618034f);
    const XMVECTOR sqrt5 =
        XMVectorReplicate(2.236068f);

    // phi^n (SIMD exp via log/exp)
    XMVECTOR phi_n = XMVectorPow(phi, n);

    // round(phi^n / sqrt5)
    return XMVectorRound(
        XMVectorDivide(phi_n, sqrt5));
}

// Compress 4 floats to 8 bytes (2x)
XMHALF4 PackToHalf4(XMVECTOR v) {
    XMHALF4 h;
    XMStoreHalf4(&h, v);
    return h;  // 8 bytes vs 16 bytes
}

HLSL Compute Shader

// GPU: each thread computes 4 values
RWStructuredBuffer<uint> buf : register(u0);

[numthreads(64, 1, 1)]
void CS_Fib(uint3 id : SV_DispatchThreadID) {
    uint b = id.x * 4;
    float4 n = float4(b, b+1, b+2, b+3);

    // HLSL auto-vectorises float4 ops
    float4 fib = round(
        pow(1.618034f, n) / 2.236068f);

    // f32tof16: pack 4 floats -> 2 uints
    uint2 pk;
    pk.x = f32tof16(fib.x)
         | (f32tof16(fib.y) << 16);
    pk.y = f32tof16(fib.z)
         | (f32tof16(fib.w) << 16);

    buf[id.x * 2]     = pk.x;
    buf[id.x * 2 + 1] = pk.y;
}

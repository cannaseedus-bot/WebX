4-wide XMVECTOR
#include <DirectXMath.h>
using namespace DirectX;

// 4 Fibonacci values in one SIMD instruction
XMVECTOR FibonacciSIMD(XMVECTOR n) {
    XMVECTOR phi   = XMVectorReplicate(1.618034f);
    XMVECTOR sqrt5 = XMVectorReplicate(2.236068f);
    return XMVectorRound(
        XMVectorDivide(XMVectorPow(phi, n), sqrt5));
}

// Pack 4 floats → XMHALF4 (8 bytes, 50% saving)
XMHALF4 PackHalf4(XMVECTOR v) {
    XMHALF4 h; XMStoreHalf4(&h, v); return h;
}

void ComputeBatch(uint32_t* idx, size_t n, XMHALF4* out) {
    for (size_t i = 0; i < n; i += 4) {
        XMVECTOR ni = XMVectorSet(
            idx[i],idx[i+1],idx[i+2],idx[i+3]);
        out[i/4] = PackHalf4(FibonacciSIMD(ni));
    }
}

}
